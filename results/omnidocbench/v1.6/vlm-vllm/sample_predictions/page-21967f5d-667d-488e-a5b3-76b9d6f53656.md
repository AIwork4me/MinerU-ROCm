Lemma 2.1. We have

\[
H ^ {\mu \nu} = - h ^ {\mu \nu} + O ^ {\mu \nu} (h ^ {2}). \tag {2.4}
\]

On one hand, the Einstein-Yang-Mills system in Lorenz gauge imply that

\[
\begin{array}{l} g ^ {\lambda \mu} \nabla^ {(\mathbf {m})} _ {\lambda} \nabla^ {(\mathbf {m})} _ {\mu} A _ {\sigma} \\ = \left(\nabla^ {(\mathbf {m})} _ {\sigma} h ^ {\alpha \mu}\right) \cdot \left(\nabla^ {(\mathbf {m})} _ {\alpha} A _ {\mu}\right) \\ + \frac {1}{2} \big (\nabla^ {(\mathbf {m}) ^ {\mu}} h _ {\sigma} ^ {\nu} + \nabla_ {\sigma} ^ {(\mathbf {m})} h ^ {\nu \mu} - \nabla^ {(\mathbf {m}) ^ {\nu}} h _ {\sigma} ^ {\mu} \big) \cdot \big (\nabla_ {\mu} ^ {(\mathbf {m})} A _ {\nu} - \nabla_ {\nu} ^ {(\mathbf {m})} A _ {\mu} \big) \\ + \frac {1}{2} \left(\nabla^ {(\mathbf {m}) ^ {\mu}} h _ {\sigma} ^ {\nu} + \nabla_ {\sigma} ^ {(\mathbf {m})} h ^ {\nu \mu} - \nabla^ {(\mathbf {m}) ^ {\nu}} h _ {\sigma} ^ {\mu}\right) \cdot \left[ A _ {\mu}, A _ {\nu} \right] \\ \left. - \left(\left[ A _ {\mu}, \nabla^ {(\mathbf {m}) ^ {\mu}} A _ {\sigma} \right] + \left[ A ^ {\mu}, \nabla_ {\mu} ^ {(\mathbf {m})} A _ {\sigma} - \nabla_ {\sigma} ^ {(\mathbf {m})} A _ {\mu} \right] + \left[ A ^ {\mu}, \left[ A _ {\mu}, A _ {\sigma} \right] \right]\right)\right) \\ + O \left(h \cdot \nabla^ {(\mathbf {m})} h \cdot \nabla^ {(\mathbf {m})} A\right) + O \left(h \cdot \nabla^ {(\mathbf {m})} h \cdot A ^ {2}\right) + O \left(h \cdot A \cdot \nabla^ {(\mathbf {m})} A\right) + O \left(h \cdot A ^ {3}\right). \tag {2.5} \\ \end{array}
\]

\[

\]

where \(P\) and \(Q\) are defined as follows:

\[
\begin{array}{l} P (\nabla^ {(\mathbf {m})} _ {\mu} h, \nabla^ {(\mathbf {m})} _ {\nu} h) := \frac {1}{4} m ^ {\alpha \alpha^ {\prime}} \nabla^ {(\mathbf {m})} _ {\mu} h _ {\alpha \alpha^ {\prime}} m ^ {\beta \beta^ {\prime}} \nabla^ {(\mathbf {m})} _ {\nu} h _ {\beta \beta^ {\prime}} - m ^ {\alpha \alpha^ {\prime}} m ^ {\beta \beta^ {\prime}} \left(\partial_ {\alpha} h _ {\beta \mu} \partial_ {\beta^ {\prime}} h _ {\alpha^ {\prime} \nu} - \nabla^ {(\mathbf {m})} _ {\beta^ {\prime}} h _ {\beta \mu} \nabla^ {(\mathbf {m})} _ {\alpha} h _ {\alpha^ {\prime} \nu}\right)} \\ + m ^ {\alpha \alpha^ {\prime}} m ^ {\beta \beta^ {\prime}} \big (\nabla_ {\mu} ^ {(\mathbf {m})} h _ {\alpha^ {\prime} \beta^ {\prime}} \nabla_ {\alpha} ^ {(\mathbf {m})} h _ {\beta \nu} - \nabla_ {\alpha} ^ {(\mathbf {m})} h _ {\alpha^ {\prime} \beta^ {\prime}} \nabla_ {\mu} ^ {(\mathbf {m})} h _ {\beta \nu} \big) \\ + m ^ {\alpha \alpha^ {\prime}} m ^ {\beta \beta^ {\prime}} \left(\nabla_ {\nu} ^ {(\mathbf {m})} h _ {\alpha^ {\prime} \beta^ {\prime}} \nabla_ {\alpha} ^ {(\mathbf {m})} h _ {\beta \mu} - \nabla_ {\alpha} ^ {(\mathbf {m})} h _ {\alpha^ {\prime} \beta^ {\prime}} \nabla_ {\nu} ^ {(\mathbf {m})} h _ {\beta \mu}\right) \\ + \frac {1}{2} m ^ {\alpha \alpha^ {\prime}} m ^ {\beta \beta^ {\prime}} \big (\nabla_ {\beta^ {\prime}} ^ {(\mathbf {m})} h _ {\alpha \alpha^ {\prime}} \nabla_ {\mu} ^ {(\mathbf {m})} h _ {\beta \nu} - \nabla_ {\mu} ^ {(\mathbf {m})} h _ {\alpha \alpha^ {\prime}} \nabla_ {\beta^ {\prime}} ^ {(\mathbf {m})} h _ {\beta \nu} \big) \\ + \frac {1}{2} m ^ {\alpha \alpha^ {\prime}} m ^ {\beta \beta^ {\prime}} \big (\nabla_ {\beta^ {\prime}} ^ {(\mathbf {m})} h _ {\alpha \alpha^ {\prime}} \nabla_ {\nu} ^ {(\mathbf {m})} h _ {\beta \mu} - \nabla_ {\nu} ^ {(\mathbf {m})} h _ {\alpha \alpha^ {\prime}} \nabla_ {\beta^ {\prime}} ^ {(\mathbf {m})} h _ {\beta \mu} \big). \\ \end{array}
\]
